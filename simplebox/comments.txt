#My Canon Scanner (Lide 110) stopped working. This fixed it.
#https://askubuntu.com/questions/1140925/how-can-i-disable-usb-autosuspend-on-ubuntu-18-04

cp /etc/default/grub /etc/default/grub.old
sed -i 's/GRUB_CMDLINE_LINUX_DEFAULT="/&usbcore.autosuspend=-1 /' /etc/default/grub
update-grub
systemctl reboot

#Linux laptop 4.19.0-8-amd64 #1 SMP Debian 4.19.98-1+deb10u1 (2020-04-27) x86_64 GNU/Linux


#Before updating /etc/default/grub,  I changed parameter using command below, tested scanner and it worked. Default value is 2.
cat /sys/module/usbcore/parameters/autosuspend
echo -1 >/sys/module/usbcore/parameters/autosuspend
